#include <stdio.h>
#include <math.h>
int is_num(char c){
    if(c=='1'||c=='2'||c=='3'||c=='4'||c=='5'||c=='6'||c=='7'||c=='8'||c=='9'||c=='0')
    return 1;
    else 
    return 0;
}

typedef struct p{
    int a[250];
}p;

int main()
{
    int m=0,n=1,k,l=0;
    int arr[250];
    char S[250];
    scanf("%[^\n]s",S);
    p b[250];
    
    for(int i=0;S[i];i++){
        if(is_num(S[i])) {
        while(is_num(S[i])){
            b[m].a[n++]=S[i]-48;
            i++;
        }
        b[m].a[0]=n;
        m++;n=1;}
    }
    for(int j=0;j<m;j++){
        if(b[j].a[2]!=0 && b[j].a[2]!=1 && b[j].a[1]!=0 && b[j].a[1]!=0){
            int a=b[j].a[0],num =0;
            for(int i=1;i<a;i++){
                num=num+(pow(10,a-1-i)*b[j].a[i]);
            }
            arr[l++]=num;
        }
    }
    
  for(int i=0;i<m;i++){
        for(int j=i+1;j<m;j++){
            if(arr[i]<arr[j]){

            }
        }
    }
